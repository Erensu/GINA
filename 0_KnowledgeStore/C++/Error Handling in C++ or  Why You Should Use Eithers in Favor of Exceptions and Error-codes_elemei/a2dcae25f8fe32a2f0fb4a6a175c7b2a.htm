#include <iostream>
#include <string>
int divide(int x, int y) {
if (y == 0) {
throw std::string("Divide by zero");
}
return x / y;
}
int main() {
try {
std::cout << "4/2 " << divide(4, 2) << std::endl;
std::cout << "3/0 " << divide(3, 0) << std::endl;
} catch(std::string e) {
std::cout << e << std::endl;
}
return 0;
}
